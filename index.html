async function refreshClickUpCache() {

            const res = await axios.get(url, { headers: { 'Authorization': process.env.CLICKUP_API_KEY } });
            if (!res.data.tasks || res.data.tasks.length === 0) {
                keepGoing = false;

            keepGoing = false;
        }
    }
    console.log(`   ‚úÖ ClickUp Cache Updated: ${allTasks.length} Brands loaded.`);
    CLICKUP_CACHE = allTasks;

function findFolderLinksInMemory(brandName) {
    const task = CLICKUP_CACHE.find(t => t.name.trim().toLowerCase() === brandName.trim().toLowerCase());
    if (!task) return null;
    const internalVal = task.custom_fields.find(f => f.id === FIELD_MAP.internal)?.value;
    const memberVal = task.custom_fields.find(f => f.id === FIELD_MAP.member)?.value;
    const extractId = (link) => {
        if (!link) return null;
        if (link.includes('id=')) return link.split('id=')[1];
        const parts = link.split('/');
        return parts[parts.length - 1] || parts[parts.length - 2];
    };
    return { internalFolderId: extractId(internalVal), memberFolderId: extractId(memberVal) };
}

// A. Create Folder
async function createDriveFolder(folderName, parentId) {
    const oauth2Client = new google.auth.OAuth2(
        process.env.GOOGLE_CLIENT_ID, process.env.GOOGLE_CLIENT_SECRET, "http://localhost"
    );
    oauth2Client.setCredentials({ refresh_token: process.env.GOOGLE_REFRESH_TOKEN });
    const drive = google.drive({ version: 'v3', auth: oauth2Client });

    const fileMetadata = {
        name: folderName,
        mimeType: 'application/vnd.google-apps.folder',
        parents: [parentId]
    };
    const res = await drive.files.create({ resource: fileMetadata, fields: 'id' });
    return res.data.id;
}

// B. Upload File
async function uploadToDrive(filePath, fileName, folderId) {
    const oauth2Client = new google.auth.OAuth2(
        process.env.GOOGLE_CLIENT_ID, process.env.GOOGLE_CLIENT_SECRET, "http://localhost"
    );
    oauth2Client.setCredentials({ refresh_token: process.env.GOOGLE_REFRESH_TOKEN });
    const drive = google.drive({ version: 'v3', auth: oauth2Client });

    const fileMetadata = { name: fileName, parents: [folderId] };
    const media = { mimeType: 'video/mp4', body: fs.createReadStream(filePath) };

    const res = await drive.files.create({ resource: fileMetadata, media: media, fields: 'id, webViewLink' });
    return res.data;
}

async function checkZoom() {
    try {
        console.log(`\nüïí Watchman Scan: ${new Date().toLocaleTimeString()}`);
        
        await refreshClickUpCache(); 
        const token = await getZoomAccessToken();

        let completed = [];
        if (fs.existsSync(LOG_PATH)) completed = JSON.parse(fs.readFileSync(LOG_PATH));

        const d = new Date();
        const toDate = d.toISOString().split('T')[0];
        d.setDate(d.getDate() - 7); 
        const fromDate = d.toISOString().split('T')[0];

        // 1. GET ALL USERS
        const usersRes = await axios.get('https://api.zoom.us/v2/users?page_size=300', { 
            headers: { 'Authorization': `Bearer ${token}` } 
        });
        const users = usersRes.data.users || [];
        console.log(`   üë• Scanning ${users.length} users (from ${fromDate} to ${toDate})...`);

        // 2. SCAN EACH USER
        for (const user of users) {
            if (IGNORE_EMAILS.includes(user.email)) continue;

            try {
                const res = await axios.get(`https://api.zoom.us/v2/users/${user.id}/recordings?from=${fromDate}&to=${toDate}`, { 
                    headers: { 'Authorization': `Bearer ${token}` } 
                });

                if (!res.data.meetings || res.data.meetings.length === 0) continue;

                for (const meeting of res.data.meetings) {
                    if (completed.includes(meeting.uuid)) continue;
                    if (IGNORE_TOPICS.some(ignored => meeting.topic.includes(ignored))) continue;

                    let links = null;
                    let brand = "";
                    let isSalesEquation = false;

                    // ------------------------------------------------------------------
                    // üõ£Ô∏è PATH 1: SALES EQUATION (Strict Formatting & Subfolders)
                    // ------------------------------------------------------------------
                    if (SALES_TEAM_FOLDERS[user.email]) {
                        
                        // Rule: Must have " x " separator. Ignore random blabla.
                        if (!meeting.topic.includes(' x ')) {
                            // console.log(`      ‚ö†Ô∏è Ignoring Sales upload "${meeting.topic}" (Invalid Format)`);
                            continue;
                        }

                        console.log(`   üöÄ New Sales Equation Video: "${meeting.topic}"`);
                        
                        // Step 1: Create a NEW Sub-Folder inside the hardcoded Parent
                        const parentId = SALES_TEAM_FOLDERS[user.email];
                        console.log(`      üìÇ Creating Sub-Folder: "${meeting.topic}"...`);
                        
                        try {
                            const subFolderId = await createDriveFolder(meeting.topic, parentId);
                            links = { internalFolderId: subFolderId, memberFolderId: subFolderId };
                            isSalesEquation = true;
                            brand = "Sales Equation"; // Generic name for logging
                        } catch (err) {
                            console.error(`      ‚ùå Could not create folder: ${err.message}`);
                            continue;
                        }
                    } 
                    // ------------------------------------------------------------------
                    // üõ£Ô∏è PATH 2: STANDARD UPLOADS (ClickUp Lookup)
                    // ------------------------------------------------------------------
                    else {
                        const nameParts = meeting.topic.split(' x ');
                        if (nameParts.length < 2) continue; 

                        // Auto-detect Brand position
                        brand = nameParts[1].split('-')[0].trim();
                        if (nameParts[1].includes("Scale Session")) brand = nameParts[0].trim();

                        console.log(`   üöÄ New Standard Video: "${meeting.topic}" (User: ${user.first_name})`);
                        links = findFolderLinksInMemory(brand);
                    }

                    // ------------------------------------------------------------------
                    // üì§ UPLOAD LOGIC
                    // ------------------------------------------------------------------
                    if (!links || !links.internalFolderId) {
                        console.log(`      ‚ùå Destination missing.`);
                        await sendEmailNotification('FAIL', meeting.topic, brand, "Destination not found.");
                        continue;
                    }

                    console.log(`      ‚úÖ Destination Found! Downloading...`);

                    for (const file of meeting.recording_files) {
                        if (file.file_type !== 'MP4' && file.file_type !== 'M4A') continue;

                        const downloadUrl = `${file.download_url}?access_token=${token}`;
                        const fileExt = file.file_type === 'MP4' ? '.mp4' : '.m4a';
                        
                        // üìõ RENAMING LOGIC üìõ
                        let finalFileName = `${meeting.topic.replace(/[^a-zA-Z0-9]/g, '_')}${fileExt}`; // Default
                        
                        // IF SALES EQUATION + MP4 -> ADD DATE
                        if (isSalesEquation && file.file_type === 'MP4') {
                            const niceDate = getFormattedDate(meeting.start_time);
                            // Format: "Debion x EE Scale Session - 20 December 2025.mp4"
                            finalFileName = `${meeting.topic} - ${niceDate}${fileExt}`; 
                        }

                        const tempPath = path.join(__dirname, finalFileName);

                        try {
                            const writer = fs.createWriteStream(tempPath);
                            const streamRes = await axios({ url: downloadUrl, method: 'GET', responseType: 'stream' });
                            streamRes.data.pipe(writer);

                            await new Promise((resolve, reject) => {
                                writer.on('finish', resolve);
                                writer.on('error', reject);
                            });

                            let targetFolder = links.memberFolderId; 
                            if (file.file_type === 'MP4') targetFolder = links.internalFolderId;

                            // Travis Override
                            if (user.email === 'travis@ecommerceequation.com.au') targetFolder = links.internalFolderId;

                            if (targetFolder) {
                                console.log(`      üì§ Uploading "${finalFileName}"...`);
                                const uploadData = await uploadToDrive(tempPath, finalFileName, targetFolder);
                                
                                if (file.file_type === 'MP4') {
                                     await sendEmailNotification('SUCCESS', meeting.topic, brand, `https://drive.google.com/file/d/${uploadData.id}/view`);
                                }
                                console.log(`      üéâ Upload Complete!`);
                            }

                        } catch (err) {
                            console.error(`      ‚ùå Error: ${err.message}`);
                            await sendEmailNotification('FAIL', meeting.topic, brand, err.message);
                        } finally {
                            if (fs.existsSync(tempPath)) fs.unlinkSync(tempPath); 
                        }
                    }

                    completed.push(meeting.uuid);
                    fs.writeFileSync(LOG_PATH, JSON.stringify(completed, null, 2));
                }
            } catch (err) { /* Ignore user errors */ }
        }

    } catch (error) {
        console.error("‚ùå Watchman Error:", error.message);
    }
}

                page++;
            }
        } catch (e) {
            keepGoing = false;
        }
    }
    console.log(`   ‚úÖ ClickUp Cache Updated: ${allTasks.length} Brands loaded.`);
    CLICKUP_CACHE = allTasks;
}

function findFolderLinksInMemory(brandName) {
    const task = CLICKUP_CACHE.find(t => t.name.trim().toLowerCase() === brandName.trim().toLowerCase());
    if (!task) return null;
    const internalVal = task.custom_fields.find(f => f.id === FIELD_MAP.internal)?.value;
    const memberVal = task.custom_fields.find(f => f.id === FIELD_MAP.member)?.value;
    const extractId = (link) => {
        if (!link) return null;
        if (link.includes('id=')) return link.split('id=')[1];
        const parts = link.split('/');
        return parts[parts.length - 1] || parts[parts.length - 2];
    };
    return { internalFolderId: extractId(internalVal), memberFolderId: extractId(memberVal) };
}

async function checkZoomDebug() {

        console.log(`\nüïí DEBUG SIMULATION (NO UPLOADS): ${new Date().toLocaleTimeString()}`);
        
        await refreshClickUpCache(); 
        const token = await getZoomAccessToken();

        console.log(`   üîé Scanning ALL Recordings for TODAY: ${today}`);

        // 1. GET ALL USERS
        const usersRes = await axios.get('https://api.zoom.us/v2/users?page_size=300', { 
            headers: { 'Authorization': `Bearer ${token}` } 
        });
        const users = usersRes.data.users || [];
        console.log(`   üë• Scanning ${users.length} users...`);

        // 2. SCAN EACH USER
        for (const user of users) {
            if (IGNORE_EMAILS.includes(user.email)) continue;

            try {
                const url = `https://api.zoom.us/v2/users/${user.id}/recordings?from=${today}&to=${today}`;
                const res = await axios.get(url, { headers: { 'Authorization': `Bearer ${token}` } });

                if (!res.data.meetings || res.data.meetings.length === 0) continue;

                for (const meeting of res.data.meetings) {
                    
                    if (IGNORE_TOPICS.some(ignored => meeting.topic.includes(ignored))) continue;

                    let links = null;
                    let brand = "";
                    let isSalesEquation = false;
                    let destinationName = "";

                    // ------------------------------------------------------------------
                    // üõ£Ô∏è PATH 1: SALES EQUATION
                    // ------------------------------------------------------------------
                    if (SALES_TEAM_FOLDERS[user.email]) {
                        if (!meeting.topic.includes(' x ')) {
                            console.log(`      ‚ö†Ô∏è  Skipping "${meeting.topic}" (Sales User but No 'x')`);
                            continue;
                        }
                        
                        console.log(`\n   üü£ SALES EQUATION FOUND: "${meeting.topic}" (User: ${user.first_name})`);
                        
                        const parentId = SALES_TEAM_FOLDERS[user.email];
                        links = { internalFolderId: "SIMULATED_SUBFOLDER_ID", memberFolderId: "SIMULATED_SUBFOLDER_ID" };
                        isSalesEquation = true;
                        brand = "Sales Equation"; 
                        destinationName = `[New Subfolder in Parent: ${parentId}]`;

                        console.log(`      üìÇ [DRY RUN] Would create folder "${meeting.topic}" inside ID: ${parentId}`);
                    } 
                    // ------------------------------------------------------------------
                    // üõ£Ô∏è PATH 2: STANDARD UPLOADS
                    // ------------------------------------------------------------------
                    else {
                        const nameParts = meeting.topic.split(' x ');
                        if (nameParts.length < 2) continue; 

                        // Auto-detect Brand
                        brand = nameParts[1].split('-')[0].trim();
                        if (nameParts[1].includes("Scale Session")) brand = nameParts[0].trim();

                        console.log(`\n   üîµ STANDARD VIDEO FOUND: "${meeting.topic}" (User: ${user.first_name})`);
                        links = findFolderLinksInMemory(brand);
                        destinationName = `[ClickUp Found Folders]`;
                    }

                    if (!links || !links.internalFolderId) {
                        console.log(`      ‚ùå Brand "${brand}" missing details.`);
                        continue;
                    }

                    // ------------------------------------------------------------------
                    // üìù SIMULATE UPLOAD
                    // ------------------------------------------------------------------
                    for (const file of meeting.recording_files) {
                        if (file.file_type !== 'MP4' && file.file_type !== 'M4A') continue;
                        const fileExt = file.file_type === 'MP4' ? '.mp4' : '.m4a';

                        // üìõ RENAMING LOGIC
                        let finalFileName = `${meeting.topic.replace(/[^a-zA-Z0-9]/g, '_')}${fileExt}`;
                        
                        // Sales Date Logic
                        if (isSalesEquation && file.file_type === 'MP4') {
                            const niceDate = getFormattedDate(meeting.start_time);
                            finalFileName = `${meeting.topic} - ${niceDate}${fileExt}`; 
                        }

                        // Determine Target
                        let targetFolder = links.memberFolderId; 
                        let targetType = "Member Link";

                        if (file.file_type === 'MP4') {
                            targetFolder = links.internalFolderId;
                            targetType = "Internal Link";
                        }

                        // Travis Override
                        if (user.email === 'travis@ecommerceequation.com.au') {
                            targetFolder = links.internalFolderId;
                            targetType = "Internal Link (Travis Force)";
                        }

                        // üñ®Ô∏è THE REQUESTED OUTPUT
                        console.log(`      üìÑ ${meeting.topic} (${file.file_type}) -> Renamed to: "${finalFileName}"`);
                        console.log(`         ‚Ü≥ is uploaded to ${targetType}: ${targetFolder} (${destinationName})`);

                        // Send Email (Only for MP4 to avoid spamming debug emails)
                        if (file.file_type === 'MP4') {
                            await sendEmailNotification('SUCCESS', finalFileName, brand, `Target Folder ID: ${targetFolder}`);
                        }
                    }
                }
            } catch (err) { /* Ignore user errors */ }
        }
        console.log("\n‚úÖ DEBUG SCAN COMPLETE.\n");

    } catch (error) {
        console.error("‚ùå Watchman Error:", error.message);
    }
}
